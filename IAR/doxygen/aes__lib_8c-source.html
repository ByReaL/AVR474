00528 static void AES_MixColumn( uint8_t * column )
00529 {
00530     uint8_t result0, result1, result2, result3;
00531     uint8_t column0, column1, column2, column3;
00532     uint8_t xor;
00533 
00534     // This generates more effective code, at least
00535     // with the IAR C compiler.
00536     column0 = column[0];
00537     column1 = column[1];
00538     column2 = column[2];
00539     column3 = column[3];
00540 
00541     // Partial sums (modular addition using XOR).
00542     result0 = column1 ^ column2 ^ column3;
00543     result1 = column0 ^ column2 ^ column3;
00544     result2 = column0 ^ column1 ^ column3;
00545     result3 = column0 ^ column1 ^ column2;
00546 
00547     // Multiply column bytes by 2 modulo AES_BPOLY. 
00548     xor = 0;
00549     if (column0 & 0x80) {
00550         xor = AES_BPOLY;
00551     }
00552     column0 <<= 1;
00553     column0  ^= xor;
00554     
00555     xor = 0;
00556     if (column1 & 0x80) {
00557         xor = AES_BPOLY;
00558     }
00559     column1 <<= 1;
00560     column1  ^= xor;
00561     
00562     xor = 0;
00563     if (column2 & 0x80) {
00564         xor = AES_BPOLY;
00565     }
00566     column2 <<= 1;
00567     column2  ^= xor;
00568     
00569     xor = 0;
00570     if (column3 & 0x80) {
00571         xor = AES_BPOLY;
00572     }
00573     column3 <<= 1;
00574     column3  ^= xor;
00575 
00576     // Final sums stored into original column bytes.
00577     column[0] = result0 ^ column0 ^ column1;
00578     column[1] = result1 ^ column1 ^ column2;
00579     column[2] = result2 ^ column2 ^ column3;
00580     column[3] = result3 ^ column0 ^ column3;
00581 }
