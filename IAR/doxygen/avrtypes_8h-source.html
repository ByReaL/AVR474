00322 #if defined(__GNUC__)
00323 static inline void delay_us(double us) __attribute__((always_inline));
00324 static inline void
00325 delay_us(double us)
00326 {
00327     /*
00328      * If us is sufficiently low to be handled by _delay_loop_1(),
00329      * use this one as it is more accurate for small values.
00330      *
00331      * Math behind it: divide F_CPU by the number of clock ticks
00332      * per _delay_loop_N cycle (3 or 4 ticks per cycle), and by
00333      * 1E6 to accomodate for the microsecond scaling.
00334      */
00335     if (F_CPU / 3E6 * us < 255)
00336     {
00337         uint8_t ticks8 = F_CPU / 3E6 * us;
00338 
00339         _delay_loop_1(ticks8);
00340     }
00341     else
00342     {
00343         uint16_t ticks16 = F_CPU / 4E6 * us;
00344 
00345         _delay_loop_2(ticks16);
00346     }
00347 }
