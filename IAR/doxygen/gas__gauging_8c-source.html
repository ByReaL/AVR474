00244 int32_t GASG_RemainingCapacity(int32_t current)
00245 {
00246     uint32_t diff;
00247     
00248     // If charging return remaining capacity for standby (ie, total remaining capacity)
00249     if(current >= 0) {
00250         diff = 0;
00251 
00252     } else {
00253         // Make current positive because calculations require that
00254         current = -current;
00255         
00256         uint16_t lowerCurrent, higherCurrent;
00257         uint16_t lowerCapacity, higherCapacity;
00258         
00259         // Find the two stored current values around the specified current
00260         uint8_t i = 1;
00261         while(battParams_sram.remainingCapacityCalibrationCurrents[i] <= current && i < REMAINING_CAPACITY_TABLE_SIZE - 1) {
00262             ++i;
00263         }
00264         lowerCurrent   = battParams_sram.remainingCapacityCalibrationCurrents[i-1];
00265         higherCurrent  = battParams_sram.remainingCapacityCalibrationCurrents[i];
00266         // Note that capacity values are scaled down 8 bits in storage
00267         lowerCapacity  = battParams_sram.remainingCapacityCalibration[i-1];
00268         higherCapacity = battParams_sram.remainingCapacityCalibration[i];
00269         
00270         // Since higherCapacity and lowerCapacity is scaled down 8 bits, that have to be
00271         // scaled up to get corrent result. But do it after division to avoid overflow
00272         // TODO: this isn't very accurate, but the value is used to present to the user
00273         //       so it will probably by converted to mAh (or minutes left) and then this
00274         //       inaccuracy probably won't be a big deal.
00275         diff = ((uint32_t)lowerCapacity << 8) + ((uint32_t)(((higherCapacity - lowerCapacity)*(current - lowerCurrent)) / (higherCurrent - lowerCurrent)) << 8);
00276     }
00277         
00278     return CCGASG_ReadAccumulatedCC() - diff;
00279 }
