00057 /* *****************************************************
00058  *
00059  * Timer Usage:
00060  *
00061  * T0: serves several functions...
00062  *     1. basic tick rate for generic timers
00063  *     2. T0CMPA is optional LED PWM function
00064  *     3. T0CMPB is Charge FET PWM
00065  *   The clock src for T0 is main clk, 1MHz.
00066  *   Therefore, base period is 1MHz/256 = 3.9KHz = 256uS
00067  *   if prescaler = 1.
00068  *   For normal operation, we need both the LEDs on CMPA
00069  *   and the overflow as a timer tick. We will therefore
00070  *   use prescale = 8. This yields a 2.048mS tick, and
00071  *   a 100Hz update rate across all 5 LEDs.  The PWM for
00072  *   the FETs is available in this design, although it is
00073  *   not used.
00074  *
00075  * T1: not used in this design.
00076  *
00077  *
00078  * *****************************************************/

00137 /* Pin change interrupt on button input.
00138  *
00139  * This interrupt will trigger if there is a pin change on the button. This will

00141  */

00273 //This interrupt functions as the main timer tick for the code.  Each time, a list
00274 // of generic timers is decremented.  Whenever one reaches zero, its expiration function
00275 // is called.  As the prescaler for T1 is initialized to 8, this fires every 2.048mS.
00276 #pragma vector = TIMER1_OVF_vect
00277 __interrupt void T1OVF_ISR(void)
00278 {
00279   unsigned int temp;
00280   unsigned char i;
00281 
00282   for(i=0; i<8; i++)
00283   {
00284     temp = generictimer[i];
00285     if(0 != temp)
00286     {
00287       temp--;
00288       generictimer[i] = temp;
00289       if(temp == 0) //when it hits 0, flag it!
00290         (GenericExpire[i])();
00291     }
00292   }
