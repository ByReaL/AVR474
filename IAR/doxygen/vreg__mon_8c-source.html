00099 __interrupt void VREGMON_ISR(void)
00100 {
00101     /* Check that VREGMON condition still exists as interrupt can't wake the
00102        device from power save sleep and interrupt flag will thus remain and
00103        trigger an interrupt even if the condition has disappeared before the
00104        device wakes up. */
00105     if ( VREGMON_ConditionExists() ) {
00106         // Save which external interrupts are enabled and disable all.
00107         SaveVregmonVar[EXTINT_SAVE] = EIMSK;
00108         EIMSK = 0x00;
00109         // Save which battery protection interrupts are enabled and disable all.
00110         SaveVregmonVar[BATPROT_SAVE] = BPIMSK;
00111         BPIMSK = 0x00;
00112         // Save which Timer0 interrupts are enabled and disable all.
00113         SaveVregmonVar[TIMER0_SAVE] = TIMSK0;
00114         TIMSK0 = 0x00;
00115         // Save which Timer1 interrupts are enabled and disable all.
00116         SaveVregmonVar[TIMER1_SAVE] = TIMSK1;
00117         TIMSK1 = 0x00;
00118         // Save SPI interrupt (if enabled) and disable it.
00119         SaveVregmonVar[SPI_SAVE] = SPCR;
00120         SPCR = SaveVregmonVar[SPI_SAVE] & ~(1<<SPIE);
00121 
00122         // Disable VADC if running. (150uA power consumption.)
00123         SaveVregmonVar[VADC_SAVE] = VADCSR;
00124         VADCSR = (1<<VADCCIF);  // Clear interrupt flag and rest if VADCSR.
00125 
00126         // Disable CCADC interrupts, but let it continue to run.
00127         SaveVregmonVar[CCADC_SAVE] = CADCSRB & ( (1<<CADACIE) | (1<<CADRCIE) | (1<<CADICIE) );
00128         CADCSRB = 0x00;
00129 
00130         // Disable EEPROM interrupt.
00131         SaveVregmonVar[EEPROM_SAVE] = EECR & (1<<EERIE);
00132         EECR = 0x00;
00133 
00134         // Save WDT configuration.
00135         SaveVregmonVar[WDT_SAVE] = WDTCSR;
00136         WDT_ResetTimer();
00137         WDT_SetTimeOut( WDTO_16ms );
00138         WDT_EnableWakeUpInterrupt();
00139 
00140         /* Warning: The following code enables *nested* interrupts. It waits
00141            for a WDT timeout to wake up the device, and then checks that the
00142            vregmon condition has really disappeared. If not it just goes back
00143            to sleep and waits for the watchdog to reset the device. */
00144         __enable_interrupt();
00145         while ( VREGMON_ConditionExists() ) {
00146             SLEEP_ENTER_POWER_SAVE();
00147         };
00148         __disable_interrupt();
00149         /* Nested interrupts disabled.*/
00150 
00151         // Set WDT to original settings.
00152         WDT_SetTimeOut( (WDT_timeout_t)(SaveVregmonVar[WDT_SAVE]  & WDT0_mask) );
00153         WDTCSR = SaveVregmonVar[WDT_SAVE];  // Possibly re-enable wdt interrupt.
00154 
00155         // Restore PRR, interrupts and modules.
00156         EIMSK = SaveVregmonVar[EXTINT_SAVE];
00157         BPIMSK = SaveVregmonVar[BATPROT_SAVE];
00158         TIMSK0 = SaveVregmonVar[TIMER0_SAVE];
00159         TIMSK1 = SaveVregmonVar[TIMER1_SAVE];
00160         SPCR = SaveVregmonVar[SPI_SAVE];
00161         VADCSR = SaveVregmonVar[VADC_SAVE];
00162         CADCSRB = SaveVregmonVar[CCADC_SAVE];
00163         EECR = SaveVregmonVar[EEPROM_SAVE];
00164 
00165         vregmon_count++; // Flag that a VREGMON condition has happened.
00166     } else {
00167         /* VREGMON condition has been cleared (ROCS bit not set), so the
00168            interrupt's only purpose is to clear the flag. */
00169     }
00170 }
