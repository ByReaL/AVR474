00108 int16_t NTC_ReadTemperatureCelsius(uint16_t ui_input_value)
00109 {
00110     // Points to index of the voltage representing the closest major temp step below the last temperature
00111     // At reset it will be set to 20 degrees, as ambient temperature most likely is between 20 and 25 degrees
00112     static uint8_t uc_last_major_temp = (uint8_t)((20-NTC_MIN_TEMPERATURE)/NTC_MAJOR_STEPS);
00113     
00114     int16_t i_temperature;
00115     
00116     // If temperature is below minimum legal value, return that (minimum temperature = maximum voltage)
00117     if(ui_input_value >= ucft_voltage[NTC_MIN_TEMPERATURE_INDEX]) {
00118         return NTC_MIN_TEMPERATURE * NTC_MINOR_STEPS;
00119     }
00120     // If temperature is above maximum legal value, return that (minimum voltage = maximum temperature)
00121     if(ui_input_value <= ucft_voltage[NTC_MAX_TEMPERATURE_INDEX]) {
00122         return NTC_MAX_TEMPERATURE * NTC_MINOR_STEPS;
00123     }
00124     
00125     // Want uc_last_major_temp to point to the major temperature step just below the voltage ui_input_value represents
00126     // If voltage is higher than at uc_last_major_temp, search downwards
00127     // else, search upwards
00128     if(ui_input_value > ucft_voltage[uc_last_major_temp] ) {
00129         do {
00130             uc_last_major_temp--;
00131         } while(ui_input_value >= ucft_voltage[uc_last_major_temp]);
00132         
00133     } else {
00134         do {
00135             uc_last_major_temp++;
00136         } while(ui_input_value <= ucft_voltage[uc_last_major_temp]);
00137         
00138         uc_last_major_temp--; //Point to a lower temp/higher voltage than what ui_input_value_represents
00139     }
00140     
00141     // Requires TEMP_MIN_TEMPERATURE to be negative
00142     i_temperature = (uc_last_major_temp + NTC_MIN_TEMPERATURE/NTC_MAJOR_STEPS)*NTC_MINOR_STEPS*NTC_MAJOR_STEPS;
00143     
00144     // ui_input_value will now be between ucft_voltage[uc_last_major_temp] and ucft_voltage[uc_last_major_temp+1]
00145     // Linear interpolation to get temperature to add to i_temperature: (MinorSteps*MajorSteps*diff + roundFix)/divisor
00146     uint16_t diff = ucft_voltage[uc_last_major_temp] - ui_input_value;
00147     uint16_t divisor = ucft_voltage[uc_last_major_temp] - ucft_voltage[uc_last_major_temp+1];
00148     uint16_t roundFix = divisor>>1;
00149     
00150     i_temperature += (NTC_MAJOR_STEPS*NTC_MINOR_STEPS*diff + roundFix)/divisor;
00151     
00152     return(i_temperature);                                  // Return the converted temperature as temperature in °C * number of minor steps per °C
00153 }
