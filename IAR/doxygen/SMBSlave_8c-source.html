00089     unsigned char temp;

00091     // Enable ACKing if not explicitly disabled later in this ISR.
00092     enableACK = TRUE;
00093 
00094     // Is this the start of a protocol?
00095     if (smb.state == SMB_STATE_IDLE)
00096     {
00097         // Reset SMBus variables.
00098         smb.txLength = 0;
00099         smb.txCount = 0;
00100         smb.rxCount = 0;
00101         smb.error = FALSE;
00102 #ifdef SMB_SUPPORT_PEC
00103         smb.pec = 0;
00104 #endif //SMB_SUPPORT_PEC
00105 
00106     }
00107 
00108     // Use the TWI status information to make desicions.
00109     switch (TWSR & 0xf8)
00110     {
00111         case 0x60:      // SLA + W received, ACK returned
00112         {
00113             // State should be IDLE when SLA+W is received. If SLA+W is received
00114             // and state is not IDLE, an error has probably occured in an earlier
00115             // transmission that could not be detected at the time. Nothing can be
00116             // done to rescue the last transmission, but the SMBus driver variables
00117             // should be reset so the ongoing transmission can complete correctly.
00118           if (smb.state != SMB_STATE_IDLE)
00119           {
00120               smb.txLength = 0;
00121               smb.txCount = 0;
00122               smb.rxCount = 0;
00123               smb.error = FALSE;
00124 #ifdef SMB_SUPPORT_PEC
00125               smb.pec = 0;
00126 #endif //SMB_SUPPORT_PEC
00127           }
00128           smb.state = SMB_STATE_WRITE_REQUESTED;
00129 
00130 #ifdef SMB_SUPPORT_PEC
00131           SMBcrc(SMB_OWN_ADDRESS_W);
00132 #endif //SMB_SUPPORT_PEC
00133           break;
00134         }
00135 
00136         case 0x80:      // Previously addressed with own SLA+W, data received, ACK returned
00137         {
00138             // Store data received in receive buffer and increase receive count.
00139             temp = TWDR;
00140             smb.rxBuffer[smb.rxCount] = temp;
00141             smb.rxCount++;
00142 
00143 #ifdef SMB_SUPPORT_PEC
00144             // Calculate PEC.
00145             SMBcrc(temp);
00146 #endif
00147             // If the receive buffer is full, disable ACKing of the next data byte.

00159             break;
00160         }
00161 
00162         case 0xa0:   // P or Sr received while still addressed.
00163         {
00164             ProcessMessage(&smb);
00165             break;
00166         }
00167 
00168         case 0xa8:  // Own SLA +R received, ACK returned.

00185 #ifdef SMB_SUPPORT_PEC

00187             SMBcrc(temp);
00188 #endif
00189 
00190             break;
00191         }
00192 
00193         case 0xb8:      // Data byte in TWDR transmitted, ACK received.
00194         {
00195             if (smb.txCount == smb.txLength)
00196             {
00197 #ifdef SMB_SUPPORT_PEC
00198                 TWDR = smb.pec;     // If PEC is enabled, an ACK is OK here.
00199 #else
00200                 smb.error = TRUE;   // If PEC is disabled, an ACK here is an error.
00201 #endif //SMB_SUPPORT_PEC
00202                 enableACK = FALSE;
00203             }
00204             else
00205             {
00206                 temp = smb.txBuffer[smb.txCount];
00207                 TWDR = temp;
00208                 smb.txCount++;
00209 #ifdef SMB_SUPPORT_PEC
00210                 SMBcrc(temp);
00211 #endif //SMB_SUPPORT_PEC

00213             }
00214             break;
00215         }
00216 
00217         case 0xc0:      // Data byte in TWDR transmitted, NACK received.
00218         {
00219             if (smb.txCount != smb.txLength)
00220             {
00221                 // Error, NACK is only expected after last data byte or PEC.
00222                 smb.error = TRUE;
00223             }
00224             smb.state = SMB_STATE_IDLE;
00225             break;
00226         }
00227 
00228         case 0xc8:      // Last data byte in TWDR transmitted, ACK received.
00229         {
00230             smb.error = TRUE;
00231             smb.state = SMB_STATE_IDLE;
00232             break;
00233         }
00234 
00235         case 0x00:      // Bus error due to an illegal START or STOP condition.
00236         {
00237             TWCR |= (1 << TWSTO);
00238             smb.error = TRUE;
00239             smb.state = SMB_STATE_IDLE;
00240             break;
00241         }
00242 
00243         default:        // Unexpected TWSR value, error.
00244         {
00245 
00246             smb.error = TRUE;
00247             smb.state = SMB_STATE_IDLE;
00248             break;
00249         }
00250     }
00251 
00252     // Issue next TWI command.
00253     if (enableACK)
00254     {
00255         // Set TWEA flag and don't clear TWINT (if set) at the same time.
00256         temp = TWCR;
00257         temp |= (1 << TWEA);
00258         temp &= ~(1 << TWINT);

00263         // Clear TWEA flag and make sure that TWINT is not cleared at the same time.
00264         TWCR &= ~((1 << TWEA) | (1 << TWINT));
00265     }
