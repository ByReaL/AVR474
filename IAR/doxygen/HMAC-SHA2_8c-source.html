00163 void HMAC_SHA2(uint32_t *ul_message, uint32_t *ul_working)
00164 {
00165     uint8_t uc_temp;  // Temporary variable
00166 
00167     // If + denotes concatenation and K the key, the HMAC is calculated this way:
00168     // HMAC(m) = SHA2( (K^outpad) + SHA2( (K^inpad) + message ) )
00169     
00170     // Since the key is one block size (512 bits) the digest from hashing only
00171     // K^inpad/outpad can be precalculated and used as initial values when hashing
00172     // with the message. They are stored in ful_inSHA and ful_outSHA and can be
00173     // generated with a utility in the "tools" folder.
00174     
00175     ul_message[SHA_INPUT_BLOCK_WORDS] = 0x80000000; // Load first padding value
00176     for(uc_temp = SHA_INPUT_BLOCK_WORDS+1; uc_temp < 15; uc_temp++)       // Pad remaining words with 0 (only one word needed to store length of message)
00177     {
00178         ul_message[uc_temp] = 0;
00179     }
00180     ul_message[15] = 512+8*SHA_INPUT_BLOCK_SIZE;        // Load message length, 512 bits from the key
00181     
00182     SHA_2(ul_message , ul_working, ful_inSHA);      // Perform a SHA-2 operation on the data
00183     
00184     
00185     for(uc_temp = 0; uc_temp < 8; uc_temp++)        // Copy result of first SHA-2 conversion
00186     {
00187         ul_message[uc_temp] = ul_working[uc_temp];    // Copy outdata from the SHA-2 into the data array before second SHA-2 operation
00188     }
00189     
00190     ul_message[8] = 0x80000000;   // Load first padding value
00191     for(uc_temp = 9; uc_temp < 15; uc_temp++)       // Pad remaining words with 0 (only one word needed to store length of message)
00192     {
00193         ul_message[uc_temp] = 0;
00194     }
00195     ul_message[15] = 512+256;         // Load message length, previous digest(256bits) + 512 bits from key
00196     
00197     SHA_2(ul_message , ul_working, ful_outSHA);   // Perform final SHA-2 operation on the data
00198 }
